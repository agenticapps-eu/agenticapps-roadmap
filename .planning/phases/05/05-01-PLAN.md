ProjectSchema = z.object({ id, name, summary: z.string().nullable(), url: z.string().nullish(),
  initiativeId, status, priority, startDate, targetDate, milestones, issueCounts })
type Project = z.infer<typeof ProjectSchema>   // auto-derives planAhead — do NOT hand-edit the type
RoadmapJsonSchema = z.object({ generatedAt, initiatives: Initiative[], projects: Project[] })

field, it parses to `planAhead === true`. That is the RED→GREEN pivot for the test below.
