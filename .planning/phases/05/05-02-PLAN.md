import type { Project, Initiative } from "@/lib/roadmap/schema";

export const isScheduled = (p: Project): boolean => p.targetDate !== null;   // Phase-4 D-06

export interface Kpis { initiatives: number; projects: number; scheduled: number;
  undated: number; byPriority: Record<number, number>; byStatus: Record<string, number>; }
export function computeKpis(projects: Project[], initiativeCount: number): Kpis;

export interface InitiativeHealth { initiative: Initiative | null; /* null = Unassigned row */
  projectCount: number; scheduled: number; undated: number;
  backlog: number; started: number; done: number; }
export function rollupInitiativeHealth(projects: Project[], initiatives: Initiative[]): InitiativeHealth[];

export interface Filters { initiatives: string[]; quarter: string | null; from: string | null;
  to: string | null; statuses: string[]; priorities: number[]; }
export function decodeFilters(sp: URLSearchParams): Filters;
export function encodeFilters(filters: Filters, base?: URLSearchParams): URLSearchParams;
export function resolveRange(filters: Filters): { start: string; end: string } | null;
export function applyFilters(projects: Project[], filters: Filters,
  range: { start: string; end: string } | null): Project[];
